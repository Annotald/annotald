  8 var numtests = 0,
  9     testfailures = 0,
 10     tests = [],
 11     currentindent = 0,
 12     nextTestFails = false;
 13 
 14 function logTest(line) {
 15     $("#testMsgBox").val($("#testMsgBox").val() + line + "\n");
 16 }
 17 
 18 function testSucceed(name) {
 19     numtests++;
 20     var indent = "";
 21     for (var i = 0; i < currentindent; i++) {
 22         indent += " ";
 23     }
 24     if (nextTestFails) {
 25         logTest(indent + "Test '" + name + "' succeeded unexpectedly.");
 26         nextTestFails = false;
 27     } else {
 28         logTest(indent + "Test '" + name + "' succeeded.");
 29     }
 30 }
 31 
 32 function testFail(name, info) {
 33     numtests++;
 34     testfailures++;
 35     var indent = "";
 36     for (var i = 0; i < currentindent; i++) {
 37         indent += " ";
 38     }
 39     if (nextTestFails) {
 40         logTest(indent + "Test '" + name + "' failed expectedly: " + info);
 41         nextTestFails = false;
 42     } else {
 43         logTest(indent + "Test '" + name + "' FAILED: " + info);
 44     }
 45 }
 46 
 47 function testFailThrow(name) {
 48     numtests++;
 49     testfailures++;
 50     var indent = "";
 51     for (var i = 0; i < currentindent; i++) {
 52         indent += " ";
 53     }
 54     logTest("Test '" + name + "' threw an ERROR");
 55 }
 56 
 57 function expectEqual(name, x, y) {
 58     if (_.isEqual(x, y)) {
 59         testSucceed(name);
 60     } else {
 61         testFail(name, "expected: " + JSON.stringify(x) + ", got: " +
 62                  JSON.stringify(y));
 63     }
 64 }
 65 
 66 function expectEqualText(name, x, y) {
 67     x = x.replace(/ /g, "").replace(/\n/g, "");
 68     y = y.replace(/ /g, "").replace(/\n/g, "");
 69     expectEqual(name, x, y);
 70 }
 71 
 72 function failingTest() {
 73     nextTestFails = true;
 74 }
 75 
 76 function suite(name, suite) {
 77     logTest("Beginning suite " + name);
 78     currentindent += 2;
 79     suite.call();
 80     currentindent -= 2;
 81     logTest("End suite " + name);
 82 }

 84 function loadTrees(trees) {
 85     $.ajax("/testLoadTrees",
 86            { async: false,
 87              success: function(res) {
 88                  $("#editpane").html(res['trees']);
 89                  resetIds(true);
 90                  resetLabelClasses(false);
 91              },
 92              dataType: "json",
 93              type: "POST",
 94              data: {trees: trees}});
 95 }

 97 function selectWord(word, end) {
 98     var selnode = $("#editpane").find(".wnode").parents().filter(function () {
 99         return wnodeString($(this)) == word;
100     }).get(0);
101     if (!end) {
102         startnode = selnode;
103         endnode = undefined;
104     } else {
105         endnode = selnode;
106     }
107     updateSelection();
108 }
109 
110 function selectParent(end) {
111     if (end) {
112         endnode = $(endnode).parent().get(0);
113     } else {
114         startnode = $(startnode).parent().get(0);
115     }
116 }
117 
118 function selectNodeByLabel(label, end) {
119     var selnode = $("#editpane").find(".snode").filter(function() {
120         return getLabel($(this)) == label;
121     }).get(0);
122     if (!end) {
123         startnode = selnode;
124         endnode = undefined;
125     } else {
126         endnode = selnode;
127     }
128     updateSelection();
129 }

131 function runTests() {
132     numtests = testfailures = 0;
133     showDialogBox("Test Results", '<textarea id="testMsgBox" style="' +
134                   'width: 100%;height: 100%;"></textarea>');
135 
136     suite("Init", function () {
137         expectEqualText("initial test HTML",
138                         $("#editpane").html(),
139                         "<divclass=\"snode\"id=\"sn0\"><divclass=\"snodeIP-MAT\
140 ipnode\"data-metadata=\"{"ID":"test-01"}\"id=\"sn1\">IP-MA\
141 T<divclass=\"snodeNP-SBJ\"id=\"sn2\">NP-SBJ<divclass=\"snodeD\"id=\"sn3\">D<sp\
142 anclass=\"wnode\">This</span></div></div><divclass=\"snodeBEP\"id=\"sn4\">BEP<\
143 spanclass=\"wnode\">is</span></div><divclass=\"snodeNP-PRD\"id=\"sn5\">NP-PRD<\
144 divclass=\"snodeD\"id=\"sn6\">D<spanclass=\"wnode\">a</span></div><divclass=\"\
145 snodeN\"id=\"sn7\">N<spanclass=\"wnode\">test</span></div></div></div></div>");
146 
147         // don't use -Text here, b/c whitespace is significant to the backend
148         expectEqual("totrees is correct",
149                     toLabeledBrackets($("#editpane")),
150                     "( (IP-MAT (NP-SBJ (D This))\n(BEP is)\n(NP-PRD (D a)\n(N \
151 test))) (ID test-01))\n\n");
152     });
153 
154     suite("Metadata", function () {
155 
156         var orig = { foo: "bar", baz: "quux" };
157         var form = $(dictionaryToForm(orig));
158 
159         expectEqual("basic metadata",
160                     orig,
161                     formToDictionary(form));
162 
163         orig.blah = {one: "1", two: "2"};
164         form = $(dictionaryToForm(orig));
165 
166         expectEqual("recursive metadata",
167                     orig,
168                     formToDictionary(form));
169 
170         orig.blah.three = { apples: "tasty", apple_pie: "tastier" };
171         form = $(dictionaryToForm(orig));
172         expectEqual("more recursive metadata",
173                     orig,
174                     formToDictionary(form));
175 
176     });
177 
178     suite("Movement", function () {
179         loadTrees("( (CP-REL (WNP-1 (WD who))\n\
180 (C 0)\n\
181 (IP-SUB (IP-SUB (NP-SBJ *T*-1) (VBD danced))\n\
182 (CONJP (CONJ and)\n\
183 (IP-SUB (VBD sang))))))");
184 
185         selectWord("sang");
186         selectWord("who", true);
187         selectParent(true);
188         leafBefore();
189         expectEqualText("ATB movement doesn't copy index",
190                         getLabel($(startnode)),
191                         "NP");
192 
193     });
194 
195     suite("Coindexation", function () {
196         loadTrees("( (IP-MAT (NP-SBJ (NPR John))\n\
197 (VBP loves) (NP-OB1 (PRO himself))\n\
198 (NP *T*)))");
199         selectNodeByLabel("NP-SBJ");
200         selectNodeByLabel("NP-OB1", true);
201         function testCoindexing (cond) {
202             coIndex();
203             expectEqual(cond + " -- Coindexation works",
204                         getIndex($(startnode)),
205                         1);
206             expectEqual(cond + " -- Coindexation works 2",
207                         getIndex($(endnode)),
208                         1);
209             expectEqual(cond + " -- Coindexation gives same index",
210                         getIndex($(startnode)),
211                         getIndex($(endnode)));
212             expectEqual(cond + " -- Index types",
213                         getIndexType($(startnode)) +
214                         getIndexType($(endnode)),
215                         "--");
216             coIndex();
217             expectEqual(cond + " -- Type cycling 1",
218                         getIndexType($(startnode)) +
219                         getIndexType($(endnode)),
220                         "-=");
221             coIndex();
222             expectEqual(cond + " -- Type cycling 2",
223                         getIndexType($(startnode)) +
224                         getIndexType($(endnode)),
225                         "=-");
226             coIndex();
227             expectEqual(cond + " -- Type cycling 3",
228                         getIndexType($(startnode)) +
229                         getIndexType($(endnode)),
230                         "==");
231             coIndex();
232             expectEqual(cond + " -- Index removal",
233                         getIndex($(startnode)),
234                         -1);
235             expectEqual(cond + " -- Index removal 2",
236                         getIndex($(endnode)),
237                         -1);
238         }
239         testCoindexing("Two phrases");
240         selectNodeByLabel("NP", true);
241         testCoindexing("Phrase and trace");
242         coIndex();
243         expectEqual("Index goes to trace leaf",
244                     wnodeString($(endnode)),
245                     "*T*-1");
246     });
247 
248     suite("POS tag validation", function () {
249         testValidLeafLabel = basesAndDashes(["FOO","BAR","BAZ"],
250                                             ["AAA","BBB","CCC"]);
251         expectEqual("correct label",
252                     testValidLeafLabel("FOO-AAA-CCC"), true);
253         expectEqual("incorrect label",
254                     testValidLeafLabel("NP-SBJ"), false);
255     });
256 
257     logTest("");
258     logTest("Test results: " + (numtests - testfailures) + "/" + numtests +
259             " passed (" +
260             Math.round(100 * (numtests - testfailures) / numtests) + "%)");
261 }
