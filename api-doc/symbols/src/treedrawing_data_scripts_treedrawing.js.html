112 Array.prototype.unique = function() {
113     var o = {}, i, l = this.length, r = [];
114     for(i=0; i<l;i+=1) o[this[i]] = this[i];
115     for(i in o) r.push(o[i]);

239 /**
240  * Add a keybinding command.
241  *
242  * Calls to this function should be in the `settings.js` file, grouped in a
243  * function called `customCommands`
244  *
245  * @param {Object} dict a mapping of properties of the keybinding.  Can
246  * contain:
247  * - `keycode`: the numeric keycode for the binding (mandatory)
248  * - `shift`: true if this is a binding with shift pressed (optional)
249  * - `ctrl`: true if this is a binding with control pressed (optional)
250  *
251  * @param {Function} fn the function to associate with the keybinding.  Any
252  * further arguments to the `addCommand` function are passed to `fn` on each
253  * invocation.
254  */
255 function addCommand(dict, fn) {
256     var commandMap;
257     if (dict.ctrl) {
258         commandMap = ctrlKeyMap;
259     } else if (dict.shift) {
260         commandMap = shiftKeyMap;
261     } else {
262         commandMap = regularKeyMap;
263     }
264     commandMap[dict.keycode] = {
265         func: fn,
266         args: Array.prototype.slice.call(arguments, 2)
267     };
268 }

289 function handleKeyDown(e) {
290     if ((e.ctrlKey && e.shiftKey) || e.metaKey || e.altKey) {
291         // unsupported modifier combinations
292         return true;
293     }
294     var commandMap;
295     if (e.ctrlKey) {
296         commandMap = ctrlKeyMap;
297     } else if (e.shiftKey) {
298         commandMap = shiftKeyMap;
299     } else {
300         commandMap = regularKeyMap;
301     }
302     last_event_was_mouse = false;
303     if (!commandMap[e.keyCode]) {
304         return true;
305     }
306     e.preventDefault();
307     var theFn = commandMap[e.keyCode].func;
308     var theArgs = commandMap[e.keyCode].args;
309     theFn.apply(undefined, theArgs);
310     if (!theFn.async) {
311         undoBarrier();
312     }
313     return false;
314 }

316 function handleNodeClick(e) {
317     e = e || window.event;
318     var element = (e.target || e.srcElement);
319     saveMetadata();
320     if (e.button == 2) {
321         // rightclick
322         if (startnode && !endnode) {
323             if (startnode != element) {
324                 e.stopPropagation();
325                 moveNode(element);
326             } else {
327                 showContextMenu(e);
328             }
329         } else if (startnode && endnode) {
330             e.stopPropagation();
331             moveNodes(element);
332         } else {
333             showContextMenu(e);
334         }
335     } else {
336         // leftclick
337         hideContextMenu();
338         if (e.shiftKey && startnode) {
339             endnode = element;
340             updateSelection();
341             e.preventDefault(); // Otherwise, this sets the text
342                                 // selection in the browser...
343         } else {
344             selectNode(element);
345             if (e.ctrlKey) {
346                 makeNode("XP");
347             }
348         }
349     }
350     e.stopPropagation();
351     last_event_was_mouse = true;
352     undoBarrier();
353 }

463 /**
464  * Select a node, and update the GUI to reflect that.
465  *
466  * @param {DOM Node} node the node to be selected
467  */
468 function selectNode(node) {
469     if (node) {
470         if (!(node instanceof Node)) {
471             try {
472                 throw Error("foo");
473             } catch (e) {
474                 console.log("selecting a non-node: " + e.stack);
475             }
476         }
477         if (node == document.getElementById("sn0")) {
478             clearSelection();
479             return;
480         }
481 
482         if (node.className == "wnode") {
483             node = node.parentNode;
484         }
485 
486         if (node == startnode) {
487             startnode = null;
488             if (endnode) {
489                 startnode = endnode;
490                 endnode = null;
491             }
492         } else if (startnode == null) {
493             startnode = node;
494         } else {
495             if (last_event_was_mouse) {
496                 if (node == endnode) {
497                     endnode = null;
498                 } else {
499                     endnode = node;
500                 }
501             } else {
502                 endnode = null;
503                 startnode = node;
504             }
505         }
506         updateSelection();
507     } else {
508         try {
509             throw Error("foo");
510         } catch (e) {
511             console.log("tried to select something falsey: " + e.stack);
512         }
513     }
514 }

634 function splitWord() {
635     if (!startnode || endnode) return;
636     if (!isLeafNode($(startnode)) || isEmpty(startnode)) return;
637     touchTree($(startnode));
638     var wordSplit = wnodeString($(startnode)).split("-");
639     var origWord = wordSplit[0];
640     var origLemma = "XXX";
641     if (wordSplit.length == 2) {
642         origLemma = "@" + wordSplit[1] + "@";
643     }
644     var origLabel = getLabel($(startnode));
645     function doSplit() {
646         var words = $("#splitWordInput").val().split("@");
647         if (words.join("") != origWord) {
648             displayWarning("The two new words don't match the original.  Aborting");
649             return;
650         }
651         if (words.length != 2) {
652             displayWarning("You can only split in one place at a time.");
653             return;
654         }
655         var labelSplit = origLabel.split("+");
656         var secondLabel = "X";
657         if (labelSplit.length == 2) {
658             setLeafLabel($(startnode), labelSplit[0]);
659             secondLabel = labelSplit[1];
660         }
661         setLeafLabel($(startnode), words[0] + "@");
662         var hasLemma = $(startnode).find(".lemma").size() > 0;
663         makeLeaf(false, secondLabel, "@" + words[1]);
664         if (hasLemma) {
665             // TODO: move to something like foo@1 and foo@2 for the two pieces
666             // of the lemmata
667             addLemma(origLemma);
668         }
669         hideDialogBox();
670     }
671     var html = "Enter an at-sign at the place to split the word: \
672 <input type='text' id='splitWordInput' value='" + origWord +
673 "' /><div id='dialogButtons'><input type='button' id='splitWordButton'\
674  value='Split' /></div>";
675     showDialogBox("Split word", html, doSplit);
676     $("#splitWordButton").click(doSplit);
677     $("#splitWordInput").focus();
678 }

712 function editComment() {
713     if (!startnode || endnode) return;
714     touchTree($(startnode));
715     var commentRaw = $.trim(wnodeString($(startnode)));
716     var commentType = commentRaw.split(":")[0];
717     // remove the {

719     var commentText = commentRaw.split(":")[1];
720     commentText = commentText.substring(0, commentText.length - 1);
721     // regex because string does not give global search.
722     commentText = commentText.replace(/_/g, " ");
723     showDialogBox("Edit Comment",
724                   '<textarea id="commentEditBox">' +
725                   commentText + '</textarea><div id="commentTypes">' +
726                   commentTypeCheckboxes + '</div><div id="dialogButtons">' +

729     $("input:radio[name=commentType]").val([commentType]);
730     $("#commentEditBox").focus().get(0).setSelectionRange(commentText.length,
731                                                           commentText.length);
732     function editCommentDone (change) {
733         if (change) {
734             var newText = $.trim($("#commentEditBox").val());
735             if (/_|\n|:|\}|\{|\(|\)/.test(newText)) {
736                 // TODO(AWE): slicker way of indicating errors...
737                 alert("illegal characters in comment: illegal characters are" +
738                       " _, :, {}, (), and newline");
739                 // hideDialogBox();
740                 $("#commentEditBox").val(newText);
741                 return;
742             }
743             newText = newText.replace(/ /g, "_");
744             commentType = $("input:radio[name=commentType]:checked").val();
745             setLabelLL($(startnode).children(".wnode"),
746                        "{" + commentType + ":" + newText + "}");
747         }
748         hideDialogBox();

1347 /**
1348  * Interpret the DOM nodes comprising the search dialog.
1349  *
1350  * This function is treponsible for transforming the representation of a
1351  * search query as HTML into an executable query, and matching it against a
1352  * node.
1353  * @private
1354  *
1355  * @param {DOM node} node the search node to interpret
1356  * @param {DOM node} target the tree node to match it against
1357  * @param {Object} options search options
1358  * @returns {DOM node} `target` if it matched the query, otherwise `undefined`
1359  */
1360  
1361 function interpretSearchNode(node, target, options) {
1362     // TODO: optimize to remove jquery calls, only use regex matching if needed
1363     // TODO: make case sensitivity an option?
1364     options = options || {};
1365     var searchtype = $(node).children(".searchtype").val();
1366     var rx, hasMatch, i, j;
1367     var newTarget = $(target).children();
1368     var childSearches = $(node).children(".searchnode");
1369 
1370     if ($(node).parent().is("#searchnodes") &&
1371         !$("#searchnodes").children(".searchnode").first().is(node) &&
1372         !options['norecurse']) {
1373         // special case siblings at root level
1374         // What an ugly hack, can it be improved?
1375         newTarget = $(target).siblings();
1376         for (j = 0; j < newTarget.length; j++) {
1377             if (interpretSearchNode(node, newTarget[j], { norecurse: true })) {
1378                 return target;
1379             }
1380         }
1381     }
1382 
1383     if (searchtype == "Label") {
1384         rx = RegExp("^" + $(node).children(".searchtext").val(), "i");
1385         hasMatch = $(target).hasClass("snode") && rx.test(getLabel($(target)));
1386         if (!hasMatch) {
1387             return undefined;
1388         }
1389     } else if (searchtype == "Text") {
1390         rx = RegExp("^" + $(node).children(".searchtext").val(), "i");
1391         hasMatch = $(target).children(".wnode").length == 1 &&
1392             rx.test(wnodeString($(target)));
1393         if (!hasMatch) {
1394             return undefined;
1395         }
1396     } else if (searchtype == "Root") {
1397         newTarget = $(target);
1398     } else if (searchtype == "Or") {
1399         for (i = 0; i < childSearches.length; i++) {
1400             if (interpretSearchNode(childSearches[i], target)) {
1401                 return target;
1402             }
1403         }
1404         return undefined;
1405     } else if (searchtype == "Deep") {
1406         newTarget = $(target).find(".snode,.wnode");
1407     } else if (searchtype == "Prec") {
1408         newTarget = $(target).nextAll();
1409     }
1410 
1411     for (i = 0; i < childSearches.length; i++) {
1412         var succ = false;
1413         for (j = 0; j < newTarget.length; j++) {
1414             if (interpretSearchNode(childSearches[i], newTarget[j])) {
1415                 succ = true;
1416                 break;
1417             }
1418         }
1419         if (!succ) {
1420             return undefined;
1421         }
1422     }
1423 
1424     return target;
1425 }

1995         if (startRoot != endRoot) {
1996             return;
1997         }
1998

2000         // if both nodes already have an index
2001         if (getIndex($(startnode)) > 0 && getIndex($(endnode)) > 0) {
2002             // and if it is the same index
2003             if (getIndex($(startnode)) == getIndex($(endnode))) {
2004                 var theIndex = getIndex($(startnode));
2005                 var types = "" + getIndexType($(startnode)) +
2006                     "" + getIndexType($(endnode));
2007                 // remove it
2008 
2009                 if (types == "=-") {
2010                     removeIndex(startnode);
2011                     removeIndex(endnode);
2012                     appendExtension($(startnode), theIndex, "=");
2013                     appendExtension($(endnode), theIndex, "=");
2014                 } else if( types == "--" ){
2015                     removeIndex(endnode);
2016                     appendExtension($(endnode), getIndex($(startnode)),"=");
2017                 } else if (types == "-=") {
2018                     removeIndex(startnode);
2019                     removeIndex(endnode);
2020                     appendExtension($(startnode), theIndex,"=");
2021                     appendExtension($(endnode), theIndex,"-");
2022                 } else if (types == "==") {
2023                     removeIndex(startnode);
2024                     removeIndex(endnode);

2029         } else if (getIndex($(startnode)) == -1 && getIndex($(endnode)) > 0) {
2030             appendExtension($(startnode), getIndex($(endnode)));
2031         } else { // no indices here, so make them
2032             var index = maxIndex(startRoot) + 1;
2033             appendExtension($(startnode), index);
2034             appendExtension($(endnode), index);
2035         }
2036     }

2397 /**

2399  *
2400  * This is a worker function, wrapped by `undo` and `redo`.
2401  * @private
2402  */
2403 function doUndo(undoData) {
2404     var map = {},
2405         newTr = [],
2406         delTr = [];
2407 
2408     _.each(undoData["map"], function(v, k) {
2409         var theNode = $("#" + k);
2410         map[k] = theNode.clone();
2411         theNode.replaceWith(v);
2412     });
2413 
2414     // Add back the deleted trees before removing the new trees, just in case
2415     // the insertion point of one of these is going to get zapped.  This
2416     // shouldn't happen, though.
2417     _.each(undoData["delTr"], function(v) {
2418         var prev = v["before"];
2419         if (prev) {
2420             v["tree"].insertAfter($("#" + prev));
2421         } else {
2422             v["tree"].prependTo($("#sn0"));
2423         }
2424         newTr.push(v["tree"].attr("id"));
2425     });
2426 
2427     _.each(undoData["newTr"], function(v) {
2428         var theNode = $("#" + v);
2429         var prev = theNode.prev();
2430         if (prev.length == 0) {
2431             prev = null;
2432         }
2433         delTr.push({
2434             tree: theNode.clone(),
2435             before: prev && prev.attr("id")
2436         });
2437         theNode.remove();
2438     });
2439 
2440     return {
2441         map: map,
2442         newTr: newTr,
2443         delTr: delTr
2444     };
2445 }
