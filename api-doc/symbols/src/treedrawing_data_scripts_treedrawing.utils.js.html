 55 function objectToTree(o) {
 56     var res = "";
 57     for (var p in o) {
 58         if (o.hasOwnProperty(p)) {
 59             res += "(" + p + " ";
 60             if (typeof o[p] == "string") { // One of life's grosser hacks
 61                 res += o[p];
 62             } else {
 63                 res += objectToTree(o[p]);
 64             }
 65             res += ")";
 66         }
 67     }
 68     return res;
 69 }

 71 /**
 72  * Convert a JS disctionary to an HTML form.
 73  *
 74  * For the metadator editing code.
 75  * @private
 76  */
 77 function dictionaryToForm(dict, level) {
 78     if (!level) {
 79         level = 0;
 80     }
 81     var res = "";
 82     if (dict) {
 83         res = '<table class="metadataTable"><thead><tr><td>Key</td>' +
 84             '<td>Value</td></tr></thead>';
 85         for (var k in dict) {
 86             if (dict.hasOwnProperty(k)) {
 87                 if (typeof dict[k] == "string") {
 88                     res += '<tr class="strval" data-level="' + level +
 89                         '"><td class="key">' + '<span style="width:"' +
 90                         4*level + 'px;"></span>' + k +
 91                         '</td><td class="val"><input class="metadataField" ' +
 92                         'type="text" name="' + k + '" value="' + dict[k] +
 93                         '" /></td></tr>';
 94                 } else if (typeof dict[k] == "object") {
 95                     res += '<tr class="tabhead"><td colspan=2>' + k +
 96                         '</td></tr>';
 97                     res += dictionaryToForm(dict[k], level + 1);
 98                 }
 99             }
100         }
101         res += '</table>';
102     }
103     return res;
104 }

106 /**
107  * Convert an HTML form into a JS dictionary
108  *
109  * For the metadata editing code
110  * @private
111  */
112 function formToDictionary(form) {
113     var d = {},
114         dstack = [],
115         curlevel = 0,
116         namestack = [];
117     form.find("tr").each(function() {
118         if ($(this).hasClass("strval")) {
119             var key = $(this).children(".key").text();
120             var val = $(this).find(".val>.metadataField").val();
121             d[key] = val;
122             if ($(this).attr("data-level") < curlevel) {
123                 var new_d = dstack.pop();
124                 var next_name = namestack.pop();
125                 new_d[next_name] = d;
126                 d = new_d;
127             }
128         } else if ($(this).hasClass("tabhead")) {
129             namestack.push($(this).text());
130             curlevel = $(this).attr("data-level");
131             dstack.push(d);
132             d = {};
133         }
134     });
135     if (dstack.length > 0) {
136         var len = dstack.length;
137         for (var i = 0; i < len; i++) {
138             var new_d = dstack.pop();
139             var next_name = namestack.pop();
140             new_d[next_name] = d;
141             d = new_d;
142         }
143     }
144     return d;
145 }

192 /**
193  * Scroll to display the next place in the document matching a selector.
194  *
195  * If no matches, do nothing.
196  *
197  * @returns {Boolean} an indicator of whether scrolling was performed
198  */
199 function scrollToNext(selector) {
200     var docViewTop = $(window).scrollTop();
201     var docViewMiddle = docViewTop + $(window).height() / 2;
202     var nextError = $(selector).filter(
203         function () {
204             return $(this).offset().top > docViewMiddle;
205         }).first();
206     if (nextError.length == 1) {
207         window.scroll(0, nextError.offset().top - $(window).height() * 0.25);
208     }
209     return !!nextError; // convert to boolean and return
210 }

218  * category.

220  * @param {String} text the text to test

267 /**
268  * Test whether a node is a leaf using heuristics.
269  *
270  * This function respects the results of the `testValidLeafLabel` and
271  * `testValidPhraseLabel` functions, if these are defined.
272  *
273  * @param {DOM node} node the node to operate on
274  */
275 function guessLeafNode(node) {
276     if (typeof testValidLeafLabel   !== "undefined" &&
277         typeof testValidPhraseLabel !== "undefined") {
278         if (testValidPhraseLabel(getLabel($(node)))) {
279             return false;
280         } else if (testValidLeafLabel(getLabel($(node)))) {
281             return true;
282         } else {
283             // not a valid label, fall back to structural check
284             return isLeafNode(node);
285         }
286     } else {
287         return isLeafNode(node);
288     }
289 }

377 /**
378  * Return the index portion of a label, or -1 if no index.
379  * @private
380  *
381  * @param {String} label the label to operate on
382  */
383 function parseIndex (label) {
384     var index = -1;
385     var lastindex = Math.max(label.lastIndexOf("-"),label.lastIndexOf("="));
386     if (lastindex == -1) {
387         return -1;
388     }
389     var lastpart = parseInt(label.substr(lastindex+1));
390     if(!isNaN(parseInt(lastpart))) {
391         index = Math.max(lastpart, index);
392     }
393     if (index == 0) {
394         return -1;
395     }
396     return index;
397 }
398 
399 /**
400  * Return the non-index portion of a label.
401  * @private
402  *
403  * @param {String} label the label to operate on
404  */
405 function parseLabel (label) {
406     var index = parseIndex(label);
407     if (index > 0) {
408         var lastindex = Math.max(label.lastIndexOf("-"),
409                                  label.lastIndexOf("="));
410         var out = $.trim("" + label.substr(0,lastindex));
411         return out;
412     }
413     return $.trim(label);
414 }

416 /**
417  * Return the type of index attached to a label, either `"-"` or `"="`.
418  * @private
419  *
420  * @param {String} label the label to operate on
421  */
422 // TODO: document that this doesn't check whether there is a numerical index,
423 // or actually do the test
424 function parseIndexType(label) {
425     var lastindex = Math.max(label.lastIndexOf("-"), label.lastIndexOf("="));
426     return label.charAt(lastindex);

429 /**
430  * Return the movement index associated with a node.
431  *
432  * @param {JQuery Node} node the node to operate on
433  */
434 function getIndex(node) {
435     if (shouldIndexLeaf(node)) {
436         return parseIndex(textNode(node.children(".wnode").first()).text());
437     } else {
438         return parseIndex(getLabel(node));
439     }

442 /**
443  * Return the type of index associated with a node, either `"-"` or `"="`.
444  *
445  * @param {JQuery Node} node the node to operate on
446  */
447 // TODO: only used once, eliminate?
448 function getIndexType (node) {
449     if (getIndex(node) < 0) {
450         return -1;
451     }
452     var label;
453     if (shouldIndexLeaf(node)) {
454         label = wnodeString(node);
455     } else {
456         label = getLabel(node);
457     }
458     var lastpart = parseIndexType(label);
459     return lastpart;

462 /**
463  * Determine whether to place a movement index on the node label or the text.
464  *
465  * @param {JQuery Node} node the node to operate on
466  */
467 function shouldIndexLeaf(node) {
468     // The below check bogusly returns true if the leftmost node in a tree is
469     // a trace, even if it is not a direct daughter.  Only do the more
470     // complicated check if we are at a POS label, otherwise short circuit
471     if (node.children(".wnode").size() == 0) return false;
472     var str = wnodeString(node);
473     return (str.substring(0,3) == "*T*" ||
474             str.substring(0,5) == "*ICH*" ||
475             str.substring(0,4) == "*CL*" ||
476             $.trim(str) == "*");
477 }

497 /**
498  * Increase the value of a tree's indices by an amount
499  * @private
500  *
501  * @param {JQuery Node} tokenRoot the token to operate on
502  * @param {number} numberToAdd
503  */
504 function addToIndices(tokenRoot, numberToAdd) {
505     var ind = 1;
506     var maxindex = maxIndex(tokenRoot);
507     var nodes = tokenRoot.find(".snode,.wnode").andSelf();
508     nodes.each(function(index) {
509         var curNode = $(this);
510         var nindex = getIndex(curNode);
511         if (nindex > 0) {
512             if (shouldIndexLeaf(curNode)) {
513                 var leafText = wnodeString(curNode);
514                 leafText = parseLabel(leafText) + parseIndexType(leafText);
515                 textNode(curNode.children(".wnode").first()).text(
516                     leafText + (nindex + numberToAdd));
517             } else {
518                 var label = getLabel(curNode);
519                 label = parseLabel(label) + parseIndexType(label);
520                 label = label + (nindex + numberToAdd);
521                 setNodeLabel(curNode, label, true);
522             }
523         }
524     });
525 }

532 function changeJustLabel (oldlabel, newlabel) {
533     var label = oldlabel;
534     var index = parseIndex(oldlabel);
535     if (index > 0) {
536         label = parseLabel(oldlabel);

543 // This function takes 3 arguments: a node label with dash tags and possibly
544 // indices, a dash tag to toggle (no dash), and a list of possible extensions
545 // (in L-to-R order).  It returns a string, which is the label with
546 // transformations applied
547 function toggleStringExtension (oldlabel, extension, extensionList) {
548     if (extension[0] == "-") {
549         // temporary compatibility hack for old configs
550         extension = extension.substring(1);
551         extensionList = extensionList.map(function(s) { return s.substring(1); });
552     }
553     var index = parseIndex(oldlabel);
554     var indextype = "";
555     if (index > 0) {
556         indextype = parseIndexType(oldlabel);
557     }
558     var currentLabel = parseLabel(oldlabel);
559 
560     // The strategy here is as follows:
561     // - split the label into an array of dash tags
562     // - operate on the array
563     // - reform the array into a string
564     currentLabel = currentLabel.split("-");
565     var labelBase = currentLabel.shift();
566     var idx = currentLabel.indexOf(extension);
567 
568     if (idx > -1) {
569         // currentLabel contains extension, remove it
570         currentLabel.splice(idx, 1);
571     } else {
572         idx = extensionList.indexOf(extension);
573         if (idx > -1) {
574             // Loop through the list, stop when we pass the right spot
575             for (var i = 0; i < currentLabel.length; i++) {
576                 if (idx < extensionList.indexOf(currentLabel[i])) {
577                     break;
578                 }
579             }
580             currentLabel.splice(i, 0, extension);
581         } else {
582             currentLabel.push(extension);
583         }
584     }
585 
586     var out = labelBase;
587     if (currentLabel.length > 0) {
588         out += "-" + currentLabel.join("-");
589     }
590     if (index > 0) {
591         out += indextype;
592         out += index;
593     }
594     return out;
595 }

597 function lookupNextLabel(oldlabel, labels) {
598     // labels is either: an array, an object
599     var newlabel = null;
600     // TODO(AWE): make this more robust!
601     if (!(labels instanceof Array)) {
602         var prefix = oldlabel.split("-")[0];
603         var new_labels = labels[prefix];
604         if (!new_labels) {
605             new_labels = _.values(labels)[0];
606         }
607         labels = new_labels;
608     }
609     for (var i = 0; i < labels.length; i++) {
610         if (labels[i] == parseLabel(oldlabel)) {
611             if (i < labels.length - 1) {
612                 newlabel = labels[i + 1];
613             } else {
614                 newlabel = labels[0];
615             }
616         }
617     }
618     if (!newlabel) {
619         newlabel = labels[0];
620     }
621     newlabel = changeJustLabel(oldlabel,newlabel);
622 
623     return newlabel;
624 }
